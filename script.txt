local function isMob(model)
    return model.Name:sub(1, 1) == "."
end

local function startFly()
    local char = LP.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    if St.flyBV then St.flyBV:Destroy() end
    St.flyBV = Instance.new("BodyVelocity")
    St.flyBV.Name = "FlyBV"
    St.flyBV.MaxForce = Vector3.new(1e5, 1e5, 1e5)
    St.flyBV.Velocity = Vector3.zero
    St.flyBV.Parent = hrp
    CollectionService:AddTag(St.flyBV, "AllowedBM")
end

local function stopFly()
    if St.flyBV then
        St.flyBV:Destroy()
        St.flyBV = nil
    end
end

-- Agility spoofer functions
local function enableAgilitySpoofer()
    local live = workspace:FindFirstChild("Live")
    if not live then return end
    
    local playerModel = live:FindFirstChild(LP.Name)
    if not playerModel then return end
    
    local passiveAgility = playerModel:FindFirstChild("PassiveAgility")
    if not passiveAgility then return end
    
    originalAgilityValue = passiveAgility.Value
    passiveAgility.Value = Options.AgilitySlider.Value
    
    if agilityConnection then agilityConnection:Disconnect() end
    
    agilityConnection = RunService.Heartbeat:Connect(function()
        local currentLive = workspace:FindFirstChild("Live")
        if currentLive then
            local currentPlayerModel = currentLive:FindFirstChild(LP.Name)
            if currentPlayerModel then
                local currentPassiveAgility = currentPlayerModel:FindFirstChild("PassiveAgility")
                if currentPassiveAgility then
                    currentPassiveAgility.Value = Options.AgilitySlider.Value
                end
            end
        end
    end)
end

local function disableAgilitySpoofer()
    if agilityConnection then
        agilityConnection:Disconnect()
        agilityConnection = nil
    end
    
    local live = workspace:FindFirstChild("Live")
    if live and originalAgilityValue then
        local playerModel = live:FindFirstChild(LP.Name)
        if playerModel then
            local passiveAgility = playerModel:FindFirstChild("PassiveAgility")
            if passiveAgility then
                passiveAgility.Value = originalAgilityValue
            end
        end
    end
end

-- Noclip functions
local function enableNoclip()
    if noclipConnection then return end
    
    noclipConnection = RunService.Stepped:Connect(function()
        local char = LP.Character
        if char then
            for _, part in pairs(char:GetDescendants()) do
                if part:IsA("BasePart") and part.CanCollide then
                    part.CanCollide = false
                end
            end
        end
    end)
end

local function disableNoclip()
    if noclipConnection then
        noclipConnection:Disconnect()
        noclipConnection = nil
    end
end

-- Improved teleport function with multiple methods
local function teleportToRealm(realmName, teleportConnectionVar, isTeleportingVar)
    if isTeleportingVar then
        if teleportConnectionVar then
            teleportConnectionVar:Disconnect()
            teleportConnectionVar = nil
        end
        isTeleportingVar = false
        return false
    end
    
    isTeleportingVar = true
    
    local function findRealmTeleporter()
        local teleporter = workspace:FindFirstChild("RealmTeleporter")
        if teleporter then return teleporter end
        
        for _, child in pairs(workspace:GetChildren()) do
            if child.Name and string.find(string.lower(child.Name), string.lower(realmName)) then
                return child
            end
        end
        
        return nil
    end
    
    local function teleportViaCFrame(teleporter)
        local char = LP.Character
        if not char then return false end
        
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if not hrp then return false end
        
        hrp.CFrame = teleporter.CFrame
        return true
    end
    
    local function teleportViaTween(teleporter)
        local char = LP.Character
        if not char then return false end
        
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if not hrp then return false end
        
        local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Linear)
        local tween = TweenService:Create(hrp, tweenInfo, {CFrame = teleporter.CFrame})
        tween:Play()
        return true
    end
    
    local teleporter = findRealmTeleporter()
    if teleporter then
        print("Found teleporter:", teleporter.Name)
        teleportViaCFrame(teleporter)
        task.wait(0.1)
        teleportViaTween(teleporter)
    else
        warn("Could not find RealmTeleporter for:", realmName)
        print("Available workspace children:")
        for _, child in pairs(workspace:GetChildren()) do
            print(" -", child.Name, child.ClassName)
        end
    end
    
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not isTeleportingVar then
            if connection then
                connection:Disconnect()
            end
            return
        end
        
        local currentTeleporter = findRealmTeleporter()
        if currentTeleporter then
            teleportViaCFrame(currentTeleporter)
        end
    end)
    
    task.delay(10, function()
        if isTeleportingVar then
            isTeleportingVar = false
            if connection then
                connection:Disconnect()
            end
            print("Stopped teleporting to", realmName, "after 10 seconds")
        end
    end)
    
    return true
end

local function hpColor(pct)
    if pct > 0.5 then
        return Color3.fromRGB(math.floor(255 * (1 - pct) * 2), 255, 0)
    else
        return Color3.fromRGB(255, math.floor(255 * pct * 2), 0)
    end
end

local function createEntry(storage, model, nameColor, distColor, fillColor, hasHP)
    if storage[model] then return end

    local nameLabel = Drawing.new("Text")
    nameLabel.Visible = false
    nameLabel.Text = model.Name
    nameLabel.Size = 14
    nameLabel.Center = true
    nameLabel.Outline = true
    nameLabel.OutlineColor = Color3.fromRGB(0, 0, 0)
    nameLabel.Color = nameColor
    nameLabel.Font = Drawing.Fonts.UI

    local distLabel = Drawing.new("Text")
    distLabel.Visible = false
    distLabel.Text = "0m"
    distLabel.Size = 12
    distLabel.Center = true
    distLabel.Outline = true
    distLabel.OutlineColor = Color3.fromRGB(0, 0, 0)
    distLabel.Color = distColor
    distLabel.Font = Drawing.Fonts.UI

    local hpLabel = nil
    if hasHP then
        hpLabel = Drawing.new("Text")
        hpLabel.Visible = false
        hpLabel.Text = "100%"
        hpLabel.Size = 12
        hpLabel.Center = true
        hpLabel.Outline = true
        hpLabel.OutlineColor = Color3.fromRGB(0, 0, 0)
        hpLabel.Color = Color3.fromRGB(0, 255, 0)
        hpLabel.Font = Drawing.Fonts.UI
    end

    local highlight = Instance.new("Highlight")
    highlight.FillColor = fillColor
    highlight.OutlineColor = fillColor
    highlight.FillTransparency = 0.5
    highlight.OutlineTransparency = 0
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Enabled = false
    highlight.Parent = model

    storage[model] = { name = nameLabel, dist = distLabel, hp = hpLabel, highlight = highlight, hasHP = hasHP }
end

local function removeEntry(storage, model)
    if storage[model] then
        storage[model].name:Remove()
        storage[model].dist:Remove()
        if storage[model].hp then
            storage[model].hp:Remove()
        end
        if storage[model].highlight and storage[model].highlight.Parent then
            storage[model].highlight:Destroy()
        end
        storage[model] = nil
    end
end

local function updateLabels(storage, toggleKey, distToggleKey, hpToggleKey, highlightToggleKey, nameColorKey, distColorKey, fillColorKey, sizeKey, fillTransKey, maxDistKey, masterEnabled)
    if not masterEnabled then
        return
    end
    
    local camera = workspace.CurrentCamera
    local lpChar = LP.Character
    local lpHRP = lpChar and lpChar:FindFirstChild("HumanoidRootPart")
    
    local maxDistance = Options[maxDistKey] and Options[maxDistKey].Value or math.huge

    for model, entry in pairs(storage) do
        if not model.Parent then
            removeEntry(storage, model)
            continue
        end

        local showName      = Toggles[toggleKey] and Toggles[toggleKey].Value or false
        local showDist      = Toggles[distToggleKey] and Toggles[distToggleKey].Value or false
        local showHP        = (entry.hasHP and hpToggleKey and Toggles[hpToggleKey] and Toggles[hpToggleKey].Value) or false
        local showHighlight = Toggles[highlightToggleKey] and Toggles[highlightToggleKey].Value or false

        local hrp = model:FindFirstChild("HumanoidRootPart")
        local hum = entry.hasHP and model:FindFirstChildOfClass("Humanoid") or nil

        if not hrp then
            entry.name.Visible = false
            entry.dist.Visible = false
            if entry.hp then entry.hp.Visible = false end
            entry.highlight.Enabled = false
            continue
        end

        if lpChar and model == lpChar then
            entry.name.Visible = false
            entry.dist.Visible = false
            if entry.hp then entry.hp.Visible = false end
            entry.highlight.Enabled = false
            continue
        end

        -- For NPCs, don't require Humanoid
        if entry.hasHP and not hum then
            entry.name.Visible = false
            entry.dist.Visible = false
            if entry.hp then entry.hp.Visible = false end
            entry.highlight.Enabled = false
            continue
        end

        local distanceTooFar = false
        if lpHRP then
            local distance = (hrp.Position - lpHRP.Position).Magnitude
            distanceTooFar = distance > maxDistance
        end

        if distanceTooFar then
            entry.name.Visible = false
            entry.dist.Visible = false
            if entry.hp then entry.hp.Visible = false end
            entry.highlight.Enabled = false
            continue
        end

        entry.highlight.Enabled = showHighlight
        if showHighlight then
            local col = Options[fillColorKey].Value
            entry.highlight.FillColor = col
            entry.highlight.OutlineColor = col
            entry.highlight.FillTransparency = Options[fillTransKey].Value
        end

        local screenPos, onScreen = camera:WorldToViewportPoint(hrp.Position + Vector3.new(0, 3.2, 0))
        if not onScreen then
            entry.name.Visible = false
            entry.dist.Visible = false
            if entry.hp then entry.hp.Visible = false end
            continue
        end

        local textSize = Options[sizeKey].Value
        local lineGap = textSize + 2
        local currentY = screenPos.Y

        entry.name.Position = Vector2.new(screenPos.X, currentY)
        entry.name.Color = Options[nameColorKey].Value
        entry.name.Size = textSize
        entry.name.Visible = showName
        if showName then currentY += lineGap end

        local dist = lpHRP and math.floor((hrp.Position - lpHRP.Position).Magnitude) or 0
        entry.dist.Text = dist .. "m"
        entry.dist.Color = Options[distColorKey].Value
        entry.dist.Size = textSize - 2
        entry.dist.Position = Vector2.new(screenPos.X, currentY)
        entry.dist.Visible = showDist
        if showDist then currentY += lineGap end

        if entry.hp and hum then
            local pct = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
            entry.hp.Text = math.floor(pct * 100) .. "%"
            entry.hp.Color = hpColor(pct)
            entry.hp.Size = textSize - 2
            entry.hp.Position = Vector2.new(screenPos.X, currentY)
            entry.hp.Visible = showHP
        end
    end
end

        for model, entry in pairs(espPlayers) do
            if entry then

                entry.dist.Visible = false
                if entry.hp then entry.hp.Visible = false end
                if entry.highlight then
                    entry.highlight.Enabled = false
                end
            end
        end

    if not espMobsMasterEnabled then
        for model, entry in pairs(espMobs) do
            if entry then
                entry.name.Visible = false
                entry.dist.Visible = false
                if entry.hp then entry.hp.Visible = false end
                if entry.highlight then

            end
        end
    end
    
    if not espNPCsMasterEnabled then
        for model, entry in pairs(espNPCs) do
            if entry then
                entry.name.Visible = false
                entry.dist.Visible = false
                if entry.hp then entry.hp.Visible = false end

                end

        end
    end
end

-- Watch for NPCs in Workspace.NPCs
local function watchNPCs()
    local npcContainer = workspace:FindFirstChild("NPCs")
    if not npcContainer then
        -- Keep waiting for NPCs folder to appear
        local npcAddedConnection
        npcAddedConnection = workspace.ChildAdded:Connect(function(child)
            if child.Name == "NPCs" then
                npcContainer = child
                npcAddedConnection:Disconnect()
                
                -- Add existing NPCs (NPCs don't have HP)
                for _, npc in pairs(npcContainer:GetChildren()) do
                    if npc:IsA("Model") then
                        createEntry(espNPCs, npc,
                            Color3.fromRGB(160, 32, 240), -- Purple color for name
                            Color3.fromRGB(200, 100, 255), -- Light purple for distance
                            Color3.fromRGB(128, 0, 255), -- Dark purple for highlight
                            false) -- No HP
                    end
                end
                
                -- Watch for new NPCs
                npcContainer.ChildAdded:Connect(function(npc)
                    if npc:IsA("Model") then
                        createEntry(espNPCs, npc,
                            Color3.fromRGB(160, 32, 240),
                            Color3.fromRGB(200, 100, 255),
                            Color3.fromRGB(128, 0, 255),
                            false) -- No HP
                    end
                end)
                
                -- Watch for removed NPCs
                npcContainer.ChildRemoved:Connect(function(npc)
                    removeEntry(espNPCs, npc)
                end)
            end
        end)
        
        -- Timeout after 30 seconds
        task.delay(30, function()
            if npcAddedConnection then
                npcAddedConnection:Disconnect()
            end
        end)
        return
    end
    
    -- Add existing NPCs (NPCs don't have HP)
    for _, npc in pairs(npcContainer:GetChildren()) do
        if npc:IsA("Model") then
            createEntry(espNPCs, npc,
                Color3.fromRGB(160, 32, 240),
                Color3.fromRGB(200, 100, 255),
                Color3.fromRGB(128, 0, 255),
                false) -- No HP
        end
    end
    
    -- Watch for new NPCs
    npcContainer.ChildAdded:Connect(function(npc)
        if npc:IsA("Model") then
            createEntry(espNPCs, npc,
                Color3.fromRGB(160, 32, 240),
                Color3.fromRGB(200, 100, 255),
                Color3.fromRGB(128, 0, 255),
                false) -- No HP
        end
    end)
    
    -- Watch for removed NPCs
    npcContainer.ChildRemoved:Connect(function(npc)
        removeEntry(espNPCs, npc)
    end)
end

local function watchLive()
    local live = workspace:WaitForChild("Live", 10)
    if not live then return end

    for _, model in ipairs(live:GetChildren()) do
        if not model:IsA("Model") then continue end
        if isMob(model) then
            createEntry(espMobs, model,
                Color3.fromRGB(255, 100, 100), Color3.fromRGB(255, 180, 180),
                Color3.fromRGB(255, 50, 50), true) -- Mobs have HP
        else
            createEntry(espPlayers, model,
                Color3.fromRGB(255, 255, 255), Color3.fromRGB(180, 220, 255),
                Color3.fromRGB(50, 100, 255), true) -- Players have HP
        end
    end

    live.ChildAdded:Connect(function(child)
        if not child:IsA("Model") then return end
        if isMob(child) then
            createEntry(espMobs, child,
                Color3.fromRGB(255, 100, 100), Color3.fromRGB(255, 180, 180),
                Color3.fromRGB(255, 50, 50), true)
        else
            createEntry(espPlayers, child,
                Color3.fromRGB(255, 255, 255), Color3.fromRGB(180, 220, 255),
                Color3.fromRGB(50, 100, 255), true)
        end
    end)

    live.ChildRemoved:Connect(function(child)
        removeEntry(espPlayers, child)
        removeEntry(espMobs, child)
    end)
end

-- RenderStepped
RunService.RenderStepped:Connect(function()
    if Toggles.FlyToggle.Value then
        local char = LP.Character
        if char then
            local hrp = char:FindFirstChild("HumanoidRootPart")
            if hrp and St.flyBV then
                local cam = workspace.CurrentCamera
                if cam then
                    local dir = Vector3.zero

                    if UserInput:IsKeyDown(Enum.KeyCode.W) then dir += cam.CFrame.LookVector end
                    if UserInput:IsKeyDown(Enum.KeyCode.S) then dir -= cam.CFrame.LookVector end
                    if UserInput:IsKeyDown(Enum.KeyCode.A) then dir -= cam.CFrame.RightVector end
                    if UserInput:IsKeyDown(Enum.KeyCode.D) then dir += cam.CFrame.RightVector end
                    if UserInput:IsKeyDown(Enum.KeyCode.Space) then dir += Vector3.new(0, 1, 0) end
                    if UserInput:IsKeyDown(Enum.KeyCode.LeftControl) then dir -= Vector3.new(0, 1, 0) end

                    St.flyBV.Velocity = if dir.Magnitude > 0 then dir.Unit * flySpeed else Vector3.zero
                end
            end
        end
    end

    updateAllESP()
end)
